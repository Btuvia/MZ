# מדריך הגדרות למערכת
# ====================
# הוראות:
# 1. פתח את הקובץ שנקרא .env.local בתיקייה (אם הוא לא קיים, צור אותו).
# 2. העתק את התוכן למטה לתוך הקובץ הזה.
# 3. במקום הטקסט בעברית (כמו "הדבק כאן..."), הדבק את הקודים האמיתיים שתשיג לפי ההוראות.

# --- חלק 1: מסד נתונים (Firebase) ---
# זה הלב של המערכת. בלי זה שום דבר לא נשמר.
# 1. כנס לאתר: https://console.firebase.google.com/
# 2. צור פרויקט חדש (לחץ על "Add project").
# 3. במסך הראשי של הפרויקט, לחץ על האייקון של "Web" (נראה כמו </>).
# 4. תן שם לאפליקציה ולחץ "Register app".
# 5. יופיע לך ריבוע עם קוד. חפש שורות שמתחילות ב-apiKey, authDomain וכו'.
# 6. העתק את הערכים שבין הגרשיים למקומות המתאימים למטה:

NEXT_PUBLIC_FIREBASE_API_KEY="כאן_את_ה_apiKey"
NEXT_PUBLIC_FIREBASE_AUTH_DOMAIN="כאן_את_ה_authDomain"
NEXT_PUBLIC_FIREBASE_PROJECT_ID="כאן_את_ה_projectId"
NEXT_PUBLIC_FIREBASE_STORAGE_BUCKET="כאן_את_ה_storageBucket"
NEXT_PUBLIC_FIREBASE_MESSAGING_SENDER_ID="כאן_את_ה_messagingSenderId"
NEXT_PUBLIC_FIREBASE_APP_ID="כאן_את_ה_appId"

# --- חלק 2: בינה מלאכותית (Gemini) ---
# כדי שהמערכת תיתן המלצות חכמות.
# 1. כנס לאתר: https://aistudio.google.com/app/apikey
# 2. התחבר עם חשבון גוגל שלך.
# 3. לחץ על "Create API key".
# 4. העתק את הקוד הארוך שמתחיל ב-AIza... והדבק למטה:

GEMINI_API_KEY="כאן_את_המפתח_מגוגל"

# --- חלק 3: מיילים (Resend) - לא חובה ---
# אם אתה רוצה שהמערכת תשלח מיילים באמת.
# 1. כנס לאתר: https://resend.com/
# 2. הירשם (חינם להתחלה).
# 3. לך ל-API Keys וצור מפתח חדש.
# 4. הדבק אותו כאן:

RESEND_API_KEY="כאן_את_המפתח_מ_resend"

# --- חלק 4: סנכרון יומן (Google Calendar) - לא חובה ---
# כדי שהמערכת תסנכרן אוטומטית עם יומן גוגל בכל התחברות.
# 1. כנס לאתר: https://console.cloud.google.com/
# 2. בחר את הפרויקט שלך (או צור חדש)
# 3. לחץ על "APIs & Services" > "Credentials"
# 4. לחץ "Create Credentials" > "OAuth 2.0 Client ID"
# 5. בחר "Web application" ותן שם
# 6. ב-"Authorized redirect URIs" הוסף:
#    http://localhost:3000/api/auth/google-calendar/callback
# 7. העתק את ה-Client ID וה-Client Secret:

GOOGLE_CLIENT_ID="כאן_את_ה_client_id_מגוגל"
GOOGLE_CLIENT_SECRET="כאן_את_ה_client_secret_מגוגל"
NEXT_PUBLIC_GOOGLE_CLIENT_ID="כאן_את_אותו_client_id_שוב"

# הערה: וודא שאתה מפעיל את Google Calendar API בפרויקט:
# https://console.cloud.google.com/apis/library/calendar-json.googleapis.com

# --- הגדרות כלליות ---
NEXT_PUBLIC_APP_URL="http://localhost:3000"
